4.1. Import a shape from Reconstruct

Convert the labeled areas to labeled volumes, and save them.

11_import_dend.py
 1import sys, os, errno
 2import numpy as np
 3import h5py
 4from pyLD import *
 5
 6
 7zpitch   = 0.04 # XX um per EM slice
 8xyzpitch = 0.02 # XX um per xyz-voxel in the volume
 9
10folder = r'dxf'
11dend   = 'dend_cyan01'
12er     = 'dend_cyan01_ER'
13mito   = 'dend_cyan01_Mito'
14psd    = 'dend_cyan01_PSD'
15filename     = 'dend01.h5'
16annot_folder = 'annot_dend01'
17
18
19# Create volume
20c = CreateVolumeFromReconstruct(folder, xyzpitch, zpitch, dend)
21volumes = [c.create(d) for d in [dend, er, mito, psd]]
22
23
24# Rotate volumes into a minimal bounding box of the xz plane
25vol_dend = volumes[0]
26r = RotateVolume(vol_dend, 1)
27volumes = [r.rotate(v) for v in volumes]
28
29
30# Swap axes for the longest z direction, depending on dendrites.
31#volumes = [v.swapaxes(1, 2) for v in volumes]
32
33
34# Pad spaces for lm simulation
35volumes = [lmpad(v) for v in volumes]
36
37
38# Extract domains from a list of volumes
39vol_dend, vol_er, vol_mito, vol_psd = volumes
40
41
42# Smooth volumes
43vol_dend = smooth_volume_erosion_then_dilation(vol_dend) 
44vol_er   = smooth_volume_dilation_then_erosion(vol_er)
45vol_psd  = smooth_volume_dilation_then_erosion(vol_psd) 
46
47
48# Logical operation of volumes
49vol_not_mito = np.logical_not( vol_mito )
50vol_not_er   = np.logical_not( vol_er )
51vol_dend_not_mito_not_ER = vol_dend ^ (vol_mito | vol_er)
52
53
54# Create surface
55bound_verts, bound_faces, bound_area_per_face, bound_areas, id_face_psd = \
56	create_surface(xyzpitch, vol_dend, PSD = vol_psd, num_smoothing = 5, method_smoothing = 'laplacian')
57mito_verts, mito_faces, _, mito_areas = create_surface(xyzpitch, vol_not_mito)
58er_verts, er_faces, _, er_areas = create_surface(xyzpitch, vol_not_er)
59
60mito_areas  = mito_areas  * vol_dend_not_mito_not_er
61er_areas    = er_areas    * vol_dend_not_mito_not_er
62bound_areas = bound_areas * vol_dend_not_mito_not_er
63
64
65# Save
66with h5py.File(filename,'w') as w:
67    w['unit length per voxel (um)'] = xyzpitch
68    w['dendrite']                   = vol_dend.astype(np.uint8)
69    w['PSD']                        = vol_psd.astype(np.uint8)
70    w['mitochondrion']              = vol_mito.astype(np.uint8)
71    w['er']                         = vol_er.astype(np.uint8)
72    w['dendrite not mitochondrion not ER'] = vol_dend_not_mito_not_er
73
74    w['boundary areas in volume']   = bound_areas
75    w['boundary vertices']          = bound_verts
76    w['boundary faces']             = bound_faces
77    w['PSD ids in boundary faces']  = id_face_psd
78
79    w['mitochondrion areas in volume'] = mito_areas
80    w['mitochondrion vertices']        = mito_verts
81    w['mitochondrion faces']           = mito_faces
82
83    w['er areas in volume'] = er_areas
84    w['er vertices']        = er_verts
85    w['er faces']           = er_faces
86
87
88# Save UNI-EM annot
89bound_color = [192,192,192]
90mito_color  = [255,255,152]
91er_color    = [179,255,179]
92surfaces = {1: [bound_verts, bound_faces, bound_color],\
93			2: [mito_verts, mito_faces, mito_color],\
94			3: [er_verts, er_faces, er_color]}
95save_uniem_annotator(annot_folder, xyzpitch, (vol_dend+vol_mito+vol_er*2).astype('uint16'), surfaces)
