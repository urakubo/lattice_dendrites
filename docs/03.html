3. Tutorial 1: schematic dendrite

In this tutorial (tutorial 1), we first build a dendritic spine that is represented by a geometrical shape (ball and stick), then two types of simulation are conducted: Ca2+ influx through NMDA receptors and fluorescence recovery after photobleaching (FRAP). Through these simulations, users can understand how to build a spiny dendrite, label the spine, run simulation, and visualize the simulation results.

All tutorial programs are located in $LD_DIRECTORY/tutorial. Find the following script in the subdirectory “1”. It provides utility functions that are used in tutorial 1. The function “make_cylinder” makes a cylinder with an indicated radius and length. The function “add_shape” locates a geometric object into a volume space. The function “show_dendrite” loads and displays the surface objects of cellular boundary, postsynaptic density (PSD), mitochondrion, and endoplasmic reticulum (ER).

tut1_functions.py
 1import numpy as np
 2import h5py
 3from skimage import morphology
 4from mayavi import mlab
 5
 6def make_cylinder(radius, length, direction = 2):
 7	arrays = [morphology.disk(radius) for _ in range(length)]
 8	cylinder = np.stack(arrays, axis=0)
 9	cylinder = cylinder.swapaxes(0, direction)
10	return cylinder
11
12def add_shape(volume, object, loc_center):
13	s = np.array(object.shape)
14	c = np.floor(s/2).astype(int)
15	b = loc_center - c
16	e = b + s
17	volume[b[0]:e[0], b[1]:e[1], b[2]:e[2] ] += object
18	volume = (volume > 0).astype(np.uint8)
19	return volume
20
21def show_dendrite(input_morpho_file):
22
23	with h5py.File(input_morpho_file,'r') as f:
24		bound_v   = f['bound vertices'][()]
25		bound_f   = f['bound faces'][()]
26		PSD_ids   = f['face id psd'][()]
27		mito_v    = f['mito vertices'][()]
28		mito_f    = f['mito faces'][()]
29		er_v      = f['er vertices'][()]
30		er_f      = f['er faces'][()]
31		pitch     = f['unit length (um)'][()]
32
33	bound_v = bound_v / pitch
34	mito_v = mito_v / pitch
35	er_v   = er_v / pitch
36
37	# Plot surface mesh
38	mlab.figure(bgcolor=(1.0,1.0,1.0), size=(700,700))
39	mlab.view( 90, 90, 300, [ 50, 30, 50 ] )
40
41	mlab.triangular_mesh(mito_v[:,0] , mito_v[:,1] , mito_v[:,2] , mito_f, color=(1.0,1.0,0.6), opacity=0.6)
42	mlab.triangular_mesh(er_v[:,0] , er_v[:,1] , er_v[:,2] , er_f, color=(0.7,1.0,0.7), opacity=0.6)
43	mlab.triangular_mesh(bound_v[:,0], bound_v[:,1], bound_v[:,2], bound_f, color=(0.7,0.7,0.7), opacity=0.3)
44	mlab.triangular_mesh(bound_v[:,0], bound_v[:,1], bound_v[:,2], bound_f[PSD_ids,:], color=(1,0,0), opacity=0.3)
45
46	return mlab
