3.3. Set reactions

まずは、好きなディレクトリに移動して、sphinx-quickstartを実行します。

以下では report というドキュメント名を指定して実行しています。

31_build_photobleach.py

 2import numpy as np
 3import h5py
 4from pyLM.units import *
 5from pyLD import * 
 6
 7
 8input_morph_file = 'models/ball_and_stick.h5'
 9output_lm_file   = 'models/photobleach.lm'

11ext  = 'default'

13er   = 'er'

15domains = {ext: 0, cyt: 1, er: 2, mito: 3}
16
17
18class SetMolecules:
19    def __init__(self, cell):
20        self.name_yfp = 'YFP'
21        self.cell     = cell
22        self.cell.define_species(self.name_yfp)
23
24    def add_molecules(self, domain_name):
25		conc_yfp = 1 # uM
26		self.cell.add_molecule_uM(self.name_yfp, conc_yfp, domain_name)
27
28    def set_diffusion(self, domain_name):
29		d_yfp    = 1 * 1e-12 # (um2/s) Kang 2012; Traffic 13:1589-1600
30		self.cell.set_diffusion(self.name_yfp, d_yfp, domain_name)
31
32    def set_reactions(self, domain_name):
33		pass
34
35
36print('\nLoad geometry data.\n')
37with h5py.File(input_morph_file,'r') as r:
38    dendrite_not_mitochondrion_not_ER = r['dendrite not mitochondrion not ER'][()]
39    pitch          = r['unit length per voxel (um)'][()]
40    PSD            = r['PSD'][()]
41    vol_bound      = r['boundary areas in volume'][()]
42
43volume = (dendrite_not_mitochondrion_not_ER > 0) * domains[cyt]
44vol_PSD  = vol_bound * (PSD > 0)
45surfaces = {'PSD': vol_PSD, 'cell boundary': vol_bound}
46
47
48print('\nBuild a lm model.\n')
49spacing_in_m = micron(pitch) # Units in SI
50cell = BuildAnyShape(sim, volume, domains, spacing_in_m, surfaces)
51
52
53print('\nSet molecules.\n')
54molecules = SetMolecules(cell)
55molecules.add_molecules(cyt)
56molecules.set_diffusion(cyt)
57molecules.set_reactions(cyt)
58
59
60print('\nSet time.\n')
61cell.sim.setTimestep(microsecond(3.0))
62cell.sim.setWriteInterval(0.05)
63cell.sim.setLatticeWriteInterval(0.05)
64cell.sim.setSimulationTime(5.0)
