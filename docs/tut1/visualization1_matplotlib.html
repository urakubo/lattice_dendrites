3.5. Visualization 1: matplotlib

Total numbers/concs of molecules were calculated in the process of simulation. The utility class ConnectTotalConcs can be used to plot concs on matplotlib.

The numbers/concs of molecules in labeled volumes (such as the spine) first need to be calculated using the class GetLabeledConcs. The caluculated numbers/concs were connected by the class ConnectLabeledConcs, which can be used to plot concs using matplotlib.

51_plot_photobleach.py
 1import os, sys
 2import numpy as np
 3from pyLD import *
 4import matplotlib.pyplot as plt
 5
 6m  = 'YFP'
 7input_label_file = 'models/labels_ball_and_stick.h5'
 8output_plot_file = 'imgs/timecourse_photobleach.png'
 9lm_files         = ['results_photobleach/0000.lm'  ,'results_photobleach/0001.lm']
10conc_files       = ['results_photobleach/c0000.h5' ,'results_photobleach/c0001.h5']
11
12# Save and connect concs of labeled spines
13s = GetLabeledConcs()
14s.load_label_volume(input_label_file)
15for (lm_file, conc_file) in zip(lm_files, conc_files):
16    s.lm_file = lm_file
17    s.exec()
18    s.save(conc_file)
19
20c = ConnectLabeledConcs(conc_files)
21
22
23# Connect total concs
24domain_ids = 1
25t = ConnectTotalConcs(lm_files, domain_ids)
26
27
28# Plot a figure
29fig = plt.figure(figsize=(6,4))
30ax  = fig.add_subplot(111)
31ax.spines['right'].set_visible(False)
32ax.spines['top'].set_visible(False)
33
34t_offset = 4
35
36ax.plot(c.timepoints-t_offset,\
37        c.get_concs( species=m, label_ids=1 ),\
38        label = m + ' in spine' )
39
40ax.plot(t.timepoints-t_offset,\
41        t.get_concs( species=m ),\
42        label = m+' in total' )
43
44# ax.set_ylim([0, 1.5])
45ax.set_xlim([-t_offset,np.max(t.timepoints-t_offset)])
46plt.xlabel('Time (s)')
47plt.ylabel('Conc (uM)')
48hans, labs = ax.get_legend_handles_labels()
49ax.legend(handles=hans, labels=labs, frameon=False)
50plt.savefig(output_plot_file, dpi=150)
51plt.show()
