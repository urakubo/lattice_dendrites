3.6. Visualization 2: mayavi

まずは、好きなディレクトリに移動して、sphinx-quickstartを実行します。

以下では report というドキュメント名を指定して実行しています。

61_video_photobleach.py
 1import os
 2import numpy as np
 3from mayavi import mlab
 4from mayavi.api import OffScreenEngine
 5from pyLD import *
 6from tut1_functions import show_dendrite
 7import subprocess as s
 8
 9# mlab.options.offscreen = True
10morpho_file = "models/ball_and_stick.h5"
11lm_files    = ['results_photobleach/0000.lm'  ,'results_photobleach/0001.lm']
12output_dir  = "imgs_for_video"
13os.makedirs(output_dir, exist_ok=True)
14
15# Show spiny dendrite
16mlab = show_dendrite(morpho_file)
17
18
19def event(lattice, sys_param, event_param):
20	id   = sys_param['id']
21	time = sys_param['time']
22	s    = sys_param['species']
23	print('Event at: {:g}, time: {:.3f}'.format(id, time))
24
25	YFP   = []
26	for i in range(lattice.shape[3]):
27		YFP.extend( np.flatnonzero(lattice[:,:,:,i] == s['YFP']).tolist() )
28	YFP = np.unravel_index(YFP, lattice[:,:,:,0].shape )
29	plot_points = mlab.points3d(YFP[0], YFP[1], YFP[2], color=(0.8,0.8,0), scale_factor=1.5,line_width=0.1)
30	text = mlab.text(0.2,0.05,"{0:.3f} s".format(time) ,color=(0,0,0), width=0.3)
31
32	mlab.view( 90, 90, 300, [ 50, 30, 50 ] )
33	fname = os.path.join(output_dir, str(id).zfill(4)+ '.png')
34	mlab.savefig(fname)
35	plot_points.remove()
36	text.remove()
37
38	return event_param
39
40print('\nGenerate images\n')
41c = ConnectAnalysis(lm_files)
42c.start_time  = -4
43c.event       = event
44c.event_param = {}
45c.exec()

49ffname = os.path.join(output_dir, '%04d'+ '.png')

51print(' '.join(com))
